# 2.20 Docker Scoutの使用
# 『コンテナセキュリティ』サンプルコード

# インストール
curl -sSfL https://raw.githubusercontent.com/docker/scout-cli/main/install.sh | sh -s --
# イメージの概要表示
docker scout quickview <イメージ名/ID>
# イメージ内のCVE表示
docker scout cves <イメージ名/ID>
# 2つのイメージを比較
docker scout compare <イメージ名/ID_A> <イメージ名/ID_B>
# ベースイメージの更新と修復推奨事項を表示
docker scout recommendations <イメージ名/ID>
