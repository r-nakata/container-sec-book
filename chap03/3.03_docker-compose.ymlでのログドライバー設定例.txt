# 3.03 docker-compose.ymlでのログドライバー設定例
# 『コンテナセキュリティ』サンプルコード

version: '3.9'

services:
  web:
    image: nginx
    logging:
      driver: "json-file"  # デフォルト、JSONファイルに保存
      options:
        max-size: "10m"
        max-file: "3"
  
  database:
    image: mysql:8.0
    logging:
      driver: "syslog"     # システムログに転送
      options:
        syslog-address: "tcp://log-server:514"
