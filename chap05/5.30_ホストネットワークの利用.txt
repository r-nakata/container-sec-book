# 5.30 ホストネットワークの利用
# 『コンテナセキュリティ』サンプルコード

# ホストネットワークでのコンテナ起動
docker run --network=host nginx

# ポート競合を避ける例
docker run --network=host -e PORT=8080 webapp1
docker run --network=host -e PORT=8081 webapp2

# 必要最小限の権限で実行する例
docker run --network=host \
  --user=1000:1000 \
  --cap-drop=ALL \
  --cap-add=NET_BIND_SERVICE \
  --read-only \
  monitoring-agent
