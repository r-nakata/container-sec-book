# 5.31 オーバーレイネットワークの使用例
# 『コンテナセキュリティ』サンプルコード

# 暗号化を有効化したoverlayネットワーク
docker network create \
  --driver overlay \
  --opt encrypted=true \
  --subnet=10.0.9.0/24 \
  --internal \
  secure_overlay

# 外部アクセスを遮断した内部専用ネットワーク
docker network create \
  --driver overlay \
  --opt encrypted=true \
  --internal \
  --attachable \
  backend_overlay
