# 5.28 セキュリティ重視アプリケーション実践例
# 『コンテナセキュリティ』サンプルコード

docker run \
  --memory=1g --memory-swap=1g \
  --cpus="0.5" \
  --pids-limit=50 \
  --ulimit nofile=1024:1024 \
  --user=1000:1000 \
  --read-only \
  --tmpfs /tmp:size=50m,noexec \
  --cap-drop=ALL \
  --log-opt max-size=10m \
  --restart=unless-stopped \
  webapp:latest
