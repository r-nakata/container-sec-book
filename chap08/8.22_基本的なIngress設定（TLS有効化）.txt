# 8.22 基本的なIngress設定（TLS有効化）
# 『コンテナセキュリティ』サンプルコード

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: api-ingress
  namespace: production
  annotations:
    nginx.ingress.kubernetes.io/ssl-redirect: "true"
spec:
  tls:
  - hosts:
    - api.example.com
    secretName: api-tls-cert
  rules:
  - host: api.example.com
    http:
      paths:
      - path: /users
        pathType: Prefix
        backend:
          service:
            name: user-service
            port:
              number: 8080
      - path: /todos
        pathType: Prefix
        backend:
          service:
            name: todo-service
            port:
              number: 8080
