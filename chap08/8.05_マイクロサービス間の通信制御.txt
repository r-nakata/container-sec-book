# 8.05 マイクロサービス間の通信制御
# 『コンテナセキュリティ』サンプルコード

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: db-service-policy
  namespace: datastore
spec:
  podSelector:
    matchLabels:
      app: mysql
  policyTypes:
  - Ingress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          app: auth-service
      namespaceSelector:
        matchLabels:
          name: backend
    - podSelector:
        matchLabels:
          app: user-profile
      namespaceSelector:
        matchLabels:
          name: backend
    ports:
    - protocol: TCP
      port: 3306
